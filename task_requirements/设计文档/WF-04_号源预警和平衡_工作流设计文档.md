## 2. 号源问题类型定义

| 问题类型 | 定义 | 严重程度 | 示例 |
|----------|------|----------|------|
| 号源为0 | 无法挂号 | 高 | 某时段号源已用完 |
| 号源紧张 | 剩余<20% | 高 | 剩余5个号源，利用率95% |
| 号源过剩 | 剩余>80% | 中 | 剩余45个号源，利用率10% |
| 号源不均 | 同科室同时段差距>50% | 中 | 张医生50个，李医生5个 |
| 工作不均 | 某医生利用率过高>85%或过低<40% | 中 | 李医生利用率92% |

### 节点3：号源统计汇总（代码节点）

| 属性 | 值 |
|------|---|
| 节点名称 | 号源统计 |
| 节点类型 | 代码节点 |
| 节点图标 | 💻 |
| 节点描述 | 统计计算和问题识别 |
| 语言 | JavaScript |
| 超时时间 | 10秒 |

#### 输入变量

| 变量 | 来源 | 类型 | 说明 |
|------|------|------|------|
| quotas | $node2 | Array | 号源数据列表 |
| threshold | $input | Float | 预警阈值 |

#### 代码逻辑

```javascript
/**
 * 号源数据统计和问题识别
 * 功能：
 * 1. 计算汇总统计
 * 2. 识别各类问题
 * 3. 计算严重程度
 */
function analyzeQuotas(input) {
    const { quotas, threshold } = input;
    const problems = [];
    
    // 汇总统计
    const stats = {
        total_schedules: quotas.length,
        total_quota: 0,
        total_used: 0,
        total_available: 0,
        avg_usage_rate: 0,
        critical_schedules: [],  // 号源为0或极低
        excess_schedules: [],    // 号源过剩
        uneven_schedules: [],    // 号源不均
        workload_issues: []     // 工作不均
    };
    
    // 存储每个时段的总号源和已用号源（用于检测不均）
    const slotTotals = {};
    
    // 按日期+时段分组
    const slotGroups = {};
    
    for (const q of quotas) {
        // 累加统计
        stats.total_quota += q.total_quota;
        stats.total_used += q.used_quota;
        stats.total_available += q.available_quota;
        
        // 创建时段分组key
        const slotKey = `${q.date}_${q.time_slot}`;
        if (!slotGroups[slotKey]) {
            slotGroups[slotKey] = [];
        }
        slotGroups[slotKey].push(q);
        
        // 识别问题
        const usageRate = q.usage_rate || (q.used_quota / q.total_quota);
        const availablePercent = q.available_quota / q.total_quota;
        
        // 问题1：号源为0（最严重）
        if (q.available_quota === 0) {
            problems.push({
                type: "号源为0",
                schedule_id: q.schedule_id,
                location: `${q.date} ${q.weekday} ${q.time_slot} ${q.physician_name}`,
                details: `剩余0个号源，无法继续挂号`,
                severity: "高",
                quota_info: {
                    total: q.total_quota,
                    used: q.used_quota,
                    available: 0
                }
            });
            stats.critical_schedules.push(q.schedule_id);
        }
        
        // 问题2：号源紧张（剩余<20%）
        else if (availablePercent < threshold) {
            const severity = usageRate > 0.9 ? "高" : "中";
            problems.push({
                type: "号源紧张",
                schedule_id: q.schedule_id,
                location: `${q.date} ${q.weekday} ${q.time_slot} ${q.physician_name}`,
                details: `剩余${q.available_quota}个号源（${(availablePercent * 100).toFixed(0)}%）`,
                severity: severity,
                quota_info: {
                    total: q.total_quota,
                    used: q.used_quota,
                    available: q.available_quota,
                    usage_rate: usageRate
                }
            });
            
            if (severity === "高") {
                stats.critical_schedules.push(q.schedule_id);
            }
        }
        
        // 问题3：号源过剩（剩余>80%）
        else if (availablePercent > 0.8) {
            problems.push({
                type: "号源过剩",
                schedule_id: q.schedule_id,
                location: `${q.date} ${q.weekday} ${q.time_slot} ${q.physician_name}`,
                details: `剩余${q.available_quota}个号源（${(availablePercent * 100).toFixed(0)}%），利用率仅${(usageRate * 100).toFixed(0)}%`,
                severity: "中",
                quota_info: {
                    total: q.total_quota,
                    used: q.used_quota,
                    available: q.available_quota,
                    usage_rate: usageRate
                }
            });
            stats.excess_schedules.push(q.schedule_id);
        }
        
        // 问题4：医生工作不均（利用率>85%或<40%）
        if (usageRate > 0.85) {
            problems.push({
                type: "工作过重",
                schedule_id: q.schedule_id,
                location: `${q.physician_name}（${q.physician_title}）`,
                details: `号源利用率${(usageRate * 100).toFixed(0)}%，过高`,
                severity: "中",
                physician_id: q.physician_id
            });
            stats.workload_issues.push(q.schedule_id);
        } else if (usageRate < 0.4 && q.total_quota > 20) {
            problems.push({
                type: "工作过轻",
                schedule_id: q.schedule_id,
                location: `${q.physician_name}（${q.physician_title}）`,
                details: `号源利用率${(usageRate * 100).toFixed(0)}%，过低`,
                severity: "低",
                physician_id: q.physician_id
            });
        }
    }
    
    // 问题5：号源不均（同一时段不同医生差距>50%）
    for (const slotKey in slotGroups) {
        const group = slotGroups[slotKey];
        if (group.length > 1) {
            const usageRates = group.map(q => q.usage_rate);
            const maxRate = Math.max(...usageRates);
            const minRate = Math.min(...usageRates);
            
            if (maxRate - minRate > 0.5) {
                const maxQ = group.find(q => q.usage_rate === maxRate);
                const minQ = group.find(q => q.usage_rate === minRate);
                
                problems.push({
                    type: "号源不均",
                    schedule_id: maxQ.schedule_id,
                    location: `${maxQ.date} ${maxQ.time_slot}时段`,
                    details: `${maxQ.physician_name}剩余${maxQ.available_quota}个，${minQ.physician_name}剩余${minQ.available_quota}个`,
                    severity: "中",
                    comparison: {
                        high: { name: maxQ.physician_name, rate: maxRate, available: maxQ.available_quota },
                        low: { name: minQ.physician_name, rate: minRate, available: minQ.available_quota }
                    }
                });
                stats.uneven_schedules.push(maxQ.schedule_id);
            }
        }
    }
    
    // 计算平均利用率
    stats.avg_usage_rate = stats.total_quota > 0 ? 
        (stats.total_used / stats.total_quota) : 0;
    
    // 判断是否需要预警
    const hasHighSeverity = problems.some(p => p.severity === "高");
    const hasMediumSeverity = problems.some(p => p.severity === "中");
    const hasIssues = problems.length > 0;
    
    return {
        stats,
        problems,
        hasIssues,
        hasHighSeverity,
        hasMediumSeverity,
        severity: hasHighSeverity ? "高" : 
                   hasMediumSeverity ? "中" : 
                   hasIssues ? "低" : "无",
        problemCount: problems.length,
        summary: {
            total: problems.length,
            high: problems.filter(p => p.severity === "高").length,
            medium: problems.filter(p => p.severity === "中").length,
            low: problems.filter(p => p.severity === "低").length
        }
    };
}

const result = analyzeQuotas({
    quotas: input.quotas,
    threshold: input.threshold
});

output.stats = result.stats;
output.problems = result.problems;
output.hasIssues = result.hasIssues;
output.hasHighSeverity = result.hasHighSeverity;
output.hasMediumSeverity = result.hasMediumSeverity;
output.severity = result.severity;
output.problemCount = result.problemCount;
output.summary = result.summary;
```

```markdown
你是一个医院号源分析专家，擅长发现号源分配不均的问题并给出调整建议。

## 你的任务

分析号源数据，找出需要关注的问题并给出平衡建议。

## 分析维度

1. **号源紧张**：剩余号源 < 20% 或绝对值 < 10个

- **高**：需要立即处理，如号源为0、高利用率(>90%)

- details: 详细描述
- severity: 严重程度（高/中/低）

- problem_index: 对应的问题索引
- suggestion: 具体的调整建议
- action: 建议的具体操作步骤
- difficulty: 操作难度（简单/中等/复杂）
- expected_effect: 预期效果

### summary（汇总）
- 主要问题：最主要的1-2个问题
```

#### 代码逻辑

```javascript
/**
 * 生成号源预警分析报告
 */
function generateReport(input) {
    const { stats, problems, suggestions, action_plans, department_name } = input;
    
    const now = new Date().toLocaleString('zh-CN');
    
    // 生成Markdown报告
    let report = `# 📊 ${department_name} 号源预警分析报告\n\n`;
    report += `生成时间：${now}\n\n`;
    
    // 1. 总体概况
    report += `## 📈 总体概况\n\n`;
    report += `| 指标 | 数值 |\n`;
    report += `|------|------|\n`;
    report += `| 总排班数 | ${stats.total_schedules} |\n`;
    report += `| 总号源 | ${stats.total_quota} |\n`;
    report += `| 已用号源 | ${stats.total_used} |\n`;
    report += `| 剩余号源 | ${stats.total_available} |\n`;
    report += `| 平均利用率 | ${(stats.avg_usage_rate * 100).toFixed(1)}% |\n`;
    report += `| 问题排班 | ${stats.critical_schedules?.length || 0} |\n\n`;
    
    // 2. 问题汇总
    if (problems.length > 0) {
        report += `## ⚠️ 问题汇总\n\n`;
        report += `共发现 **${problems.length}** 个问题：\n\n`;
        
        // 按严重程度分组
        const high = problems.filter(p => p.severity === "高");
        const medium = problems.filter(p => p.severity === "中");
        const low = problems.filter(p => p.severity === "低");
        
        if (high.length > 0) {
            report += `### 🔴 高严重程度（${high.length}个）\n\n`;
            high.forEach((p, i) => {
                report += `${i + 1}. **${p.type}** - ${p.location}\n`;
                report += `   - ${p.details}\n\n`;
            });
        }
        
        if (medium.length > 0) {
            report += `### 🟡 中严重程度（${medium.length}个）\n\n`;
            medium.forEach((p, i) => {
                report += `${i + 1}. **${p.type}** - ${p.location}\n`;
                report += `   - ${p.details}\n\n`;
            });
        }
        
        if (low.length > 0) {
            report += `### 🟢 低严重程度（${low.length}个）\n\n`;
            low.forEach((p, i) => {
                report += `${i + 1}. **${p.type}** - ${p.location}\n`;
                report += `   - ${p.details}\n\n`;
            });
        }
    }
    
    // 3. 调整建议
    if (action_plans && action_plans.length > 0) {
        report += `## 💡 调整建议\n\n`;
        action_plans.forEach((plan, i) => {
            report += `### 建议${i + 1}：${plan.action_type}\n\n`;
            report += `- 操作：从 ${plan.source} 调整到 ${plan.target}\n`;
            report += `- 号源变化：${plan.quota_change > 0 ? '+' : ''}${plan.quota_change}个\n`;
            report += `- 难度：${plan.difficulty}\n`;
            report += `- 优先级：${plan.priority}\n`;
            report += `- 预期效果：${plan.expected_result}\n\n`;
        });
    }
    
    // 4. 后续操作
    report += `## 📋 后续操作\n\n`;
    report += `- [ ] 查看详细号源数据\n`;
    report += `- [ ] 确认调整方案\n`;
    report += `- [ ] 执行号源调整\n`;
    report += `- [ ] 通知相关人员\n\n`;
    
    report += `---\n`;
    report += `*报告自动生成，请登录系统查看详情*\n`;
    
    // 生成HTML版本（简化）
    const report_html = report
        .replace(/\n/g, '<br>')
        .replace(/#{1,6} /g, '<b>')
        .replace(/\*\*([^*]+)\*\*/g, '<strong>$1</strong>');
    
    return {
        report,
        report_html,
        summary: {
            total_problems: problems.length,
            high_severity: problems.filter(p => p.severity === "高").length,
            medium_severity: problems.filter(p => p.severity === "中").length,
            low_severity: problems.filter(p => p.severity === "低").length,
            action_plans_count: action_plans?.length || 0
        }
    };
}

const result = generateReport({
    stats: input.stats,
    problems: input.problems,
    suggestions: input.suggestions,
    action_plans: input.action_plans,
    department_name: input.department_name
});

output.report = result.report;
output.report_html = result.report_html;
output.summary = result.summary;
```

## 4. 异常处理

### 4.1 各节点异常处理

| 节点 | 异常类型 | 处理方式 |
|------|----------|----------|

| 节点5 | 判断失败 | 默认继续（保守策略） |
| 节点6 | LLM调用失败 | 使用基本建议继续 |
| 节点7 | 报告生成失败 | 生成简化报告 |
| 节点8 | 发送失败 | 记录日志，不影响主流程 |
| 节点9 | 记录失败 | 记录到本地日志文件 |

### 9.3 数据库表新增

```sql
CREATE TABLE quota_alert_log (
    id              SERIAL PRIMARY KEY,
    department_id   INTEGER NOT NULL, -- 科室ID
    alert_date      DATE NOT NULL,          -- 预警日期
    days_ahead      INTEGER DEFAULT 7,     -- 预测天数
    total_schedules INTEGER DEFAULT 0,     -- 总排班数
    total_quota     INTEGER DEFAULT 0,      -- 总号源
    total_available INTEGER DEFAULT 0,     -- 剩余号源
    avg_usage_rate  DECIMAL(5,4),          -- 平均利用率
    problem_count  INTEGER DEFAULT 0,     -- 问题数量
    high_severity  INTEGER DEFAULT 0,      -- 高严重问题数
    medium_severity INTEGER DEFAULT 0,     -- 中严重问题数
    severity       VARCHAR(8) DEFAULT '无', -- 严重程度：高/中/低/无
    report         TEXT,                   -- 完整报告
    trigger_type   VARCHAR(8) DEFAULT 'auto', -- 触发类型：auto/manual
    created_time   TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    delmark        SMALLINT DEFAULT 0
);
```
